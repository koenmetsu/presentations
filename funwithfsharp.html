public class DirectoryPath
{
    private readonly string _path;

    public DirectoryPath(string path)
    {
        _path = path;
    }

    public string Path
    {
        get { return _path; }
    }

    public static implicit operator string(DirectoryPath directoryPath)
    {
        return directoryPath.Path;
    }
}
